1   package com.milo.brkut.logic;

3   import com.milo.brkut.main.Config;
4   import java.awt.Color;
5   
6   /**
7    * The Player.
8    * @author milo
9    */
10  public class Player extends GameObject {
11  
12      private int lives;
13      private boolean alive;
14  
15      public Player(double x, double y, double width, double height) {
16          super(x, y, width, height);
17          this.lives = Config.PLAYER_LIVES;
18          this.color = Color.WHITE;
19          this.alive = true;
20          this.setMaxSpeed(Config.PLAYER_MAXSPEED);
21      }
22  
23      public int getLives() {
24          return this.lives;
25      }
26  
27      /**
28       * Adds an extra life.
29       */
30      public void addLife() {
31          this.lives++;
32      }
33  
34      /**
35       * Kills the player.
36       */
37      public void kill() {
38          this.alive = false;
39          this.lives--;
40      }
41  
42      /**
43       * Respawns the player to the starting position.
44       * @param x Starting position x coordinate
45       * @param y Starting position y coordinate
46       */
47      public void respawn(double x, double y) {
48          this.alive = true;
49          this.stop();
50          this.moveTo(x, y);
51          this.setWidth(Config.PLAYER_WIDTH);
52          this.setHeight(Config.PLAYER_HEIGHT);
53          this.setColor(Config.PLAYER_COLOR);
54      }
55  
56      @Override
57      boolean isAlive() {
58          return this.alive;
59      }
60  
61  }
