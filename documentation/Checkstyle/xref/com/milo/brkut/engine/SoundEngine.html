18       * Plays the ball-hits-brick-sound.
19       */

21          play("media/hit1.wav");
22      }
23

39       * Plays the player died -sound.

44  
45      /**
46       * Tries to play a sound sample.
47       *
48       * @param sample Sample filename
49       */
50      private void play(String sample) {
51          if (Config.SOUNDS) {
52              try {
53                  InputStream in = new FileInputStream(sample);
54                  AudioStream audio = new AudioStream(in);
55                  AudioPlayer.player.start(audio);
56              } catch (Exception ex) {
57              }
58          }
59      }
60  }
